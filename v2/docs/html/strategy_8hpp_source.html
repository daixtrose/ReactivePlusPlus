28template<typename S, typename Type>
29concept operator_strategy = requires(const S& const_strategy,
30 S& strategy,
31 const Type& v,
32 Type&& mv,
33 const disposable_wrapper disposable,
34 const dynamic_observer<Type>& const_observer,
35 dynamic_observer<Type>& observer)
36{
37 const_strategy.on_subscribe(observer);
38
39 const_strategy.on_next(const_observer, v);
40 const_strategy.on_next(const_observer, std::move(mv));
41 const_strategy.on_error(observer, std::exception_ptr{});
42 const_strategy.on_completed(observer);
43
44 strategy.set_upstream(observer, disposable);
45 { strategy.is_disposed() } -> std::same_as<bool>;
46};
47}

55{
56public:

63 template<typename...Args>
64 operator_strategy_base(captured_observer&& observer, Args&&...args)
65 : m_observer{std::move(observer)}
66 , m_strategy{std::forward<Args>(args)...}
67 {
68 m_strategy.on_subscribe(m_observer);
69 }

77 ~operator_strategy_base() noexcept = default;
78
79 void set_upstream(const disposable_wrapper& d) { m_strategy.set_upstream(m_observer, d); }
80 bool is_disposed() const { return m_strategy.is_disposed() || m_observer.is_disposed(); }
81
82 void on_next(const T& v) const { m_strategy.on_next(m_observer, v); }
83 void on_next(T&& v) const { m_strategy.on_next(m_observer, std::move(v)); }
84 void on_error(const std::exception_ptr& err) const { m_strategy.on_error(m_observer, err); }
85 void on_completed() const { m_strategy.on_completed(m_observer); }
