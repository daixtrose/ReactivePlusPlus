103 template<constraint::decayed_type Type, constraint::observer_strategy<Type> Strategy, typename DisposablesStrategy>
104 class base_observer_impl
105 {
106 protected:
107  template<typename... Args>
108  requires constraint::is_constructible_from<Strategy, Args&&...>
109  explicit base_observer_impl(DisposablesStrategy&& strategy, Args&&... args)
110  : m_strategy{std::forward<Args>(args)...}, m_disposable{std::move(strategy)}
111  {
112  }
113 
114  base_observer_impl(const base_observer_impl&) = default;
115  base_observer_impl(base_observer_impl&&) noexcept = default;
116 
117 public:
121  void set_upstream(const disposable_wrapper& d)
122  {
123  m_disposable.set_upstream(d);
124  m_strategy.set_upstream(d);
125  }
126 
133  bool is_disposed() const
134  {
135  return m_disposable.is_disposed() || m_strategy.is_disposed();
136  }
142  void on_next(const Type& v) const
143  {
144  try
145  {
146  if (!is_disposed())
147  m_strategy.on_next(v);
148  }
149  catch(...)
150  {
151  on_error(std::current_exception());
152  }
153  }
154 
160  void on_next(Type&& v) const
161  {
162  try
163  {
164  if (!is_disposed())
165  m_strategy.on_next(std::move(v));
166  }
167  catch(...)
168  {
169  on_error(std::current_exception());
170  }
171  }
172 
178  void on_error(const std::exception_ptr& err) const
179  {
180  if (!is_disposed())
181  {
182  m_strategy.on_error(err);
183  m_disposable.dispose();
184  }
185  }
186 
191  void on_completed() const
192  {
193  if (!is_disposed())
194  {
195  m_strategy.on_completed();
196  m_disposable.dispose();
197  }
198  }
199 
200 private:
201  RPP_NO_UNIQUE_ADDRESS Strategy m_strategy;
202  RPP_NO_UNIQUE_ADDRESS DisposablesStrategy m_disposable;
203 };
