26template<constraint::decayed_type Container>
27class shared_container
28{
29public:
30 template<typename ...Ts>
31 requires (!constraint::variadic_decayed_same_as<shared_container<Container>, Ts...>)
32 explicit shared_container(Ts&&...items)
33 // raw "new" call to avoid extra copy/moves for items
34 : m_container{new Container{std::forward<Ts>(items)...}} {}
35
36 shared_container(const shared_container&) = default;
37 shared_container(shared_container&&) noexcept = default;
38
39 auto begin() const { return std::cbegin(*m_container); }
40 auto end() const { return std::cend(*m_container); }
41
42 auto get_actual_iterator() const
43 {
44 if (!m_iterator)
45 m_iterator.emplace(begin());
46 return m_iterator.value();
47 }
48 bool increment_iterator() const
49 {
50 if (!m_iterator)
51 m_iterator.emplace(begin());
52
53 return ++(m_iterator.value()) != end();
54 }
55
56 static const auto& extract_value_from_itr(const auto& itr) {
57 return *itr;
58 }
59
60private:
61 std::shared_ptr<Container> m_container{};
62 mutable std::optional<decltype(std::cbegin(*m_container))> m_iterator;
63};

65template<constraint::decayed_type Container>
66class container_with_iterator
67{
68public:
69 template<typename ...Ts>
70 requires (!constraint::variadic_decayed_same_as<container_with_iterator<Container>, Ts...>)
71 explicit container_with_iterator(Ts&&...items)
72 : m_container{std::forward<Ts>(items)...} {}
73
74 container_with_iterator(const container_with_iterator& other)
75 : m_container{other.m_container}
76 , m_index(other.m_index)
77 {}
78
79 container_with_iterator(container_with_iterator&& other) noexcept
80 : m_container{std::move(other.m_container)}
81 , m_index(other.m_index)
82 {}
83
84 auto begin() const { return std::cbegin(m_container); }
85 auto end() const { return std::cend(m_container); }
86
87 auto get_actual_iterator() const
88 {
89 if (!m_iterator)
90 m_iterator.emplace(get_default_iterator_value());
91 return m_iterator.value();
92 }
93 bool increment_iterator() const
94 {
95 if (!m_iterator)
96 m_iterator.emplace(get_default_iterator_value());
97
98 ++m_index;
99 return ++(m_iterator.value()) != end();
100 }
101
102 static auto extract_value_from_itr(const auto& itr) {
103 return std::move(*itr);
104 }
105
106private:
107 auto get_default_iterator_value() const
108 {
109 auto itr = begin();
110 std::advance(itr, m_index);
111 return itr;
112 }
113
114private:
115 RPP_NO_UNIQUE_ADDRESS Container m_container{};
116 mutable size_t m_index{};
117 mutable std::optional<decltype(std::cbegin(m_container))> m_iterator{};
118};

120template<constraint::memory_model memory_model, constraint::iterable Container, typename ...Ts>
121auto pack_to_container(Ts&& ...items)
122{
123 if constexpr (std::same_as<memory_model, rpp::memory_model::use_stack>)
124 return container_with_iterator<Container>{std::forward<Ts>(items)...};
125 else
126 return shared_container<Container>{std::forward<Ts>(items)...};
127}

135struct from_iterable_schedulable
136{
137 template<constraint::decayed_type PackedContainer, constraint::observer_strategy<utils::iterable_value_t<PackedContainer>> Strategy>
138 rpp::schedulers::optional_duration operator()(const observer<utils::iterable_value_t<PackedContainer>, Strategy>& obs, const PackedContainer& cont) const
139 {
140 try
141 {
142 if (const auto itr = cont.get_actual_iterator(); itr != std::cend(cont))
143 {
144 obs.on_next(utils::as_const(*itr));
145 if (cont.increment_iterator()) // it was not last
146 return schedulers::duration{}; // re-schedule this
147 }
148
149 obs.on_completed();
150 }
151 catch (...)
152 {
153 obs.on_error(std::current_exception());
154 }
155 return std::nullopt;
156 }
157};

159template<constraint::decayed_type PackedContainer, schedulers::constraint::scheduler TScheduler>
160struct from_iterable_strategy
161{
162 RPP_NO_UNIQUE_ADDRESS PackedContainer container;
163 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
164
165 template<constraint::observer_strategy<utils::iterable_value_t<PackedContainer>> Strategy>
166 void subscribe(observer<utils::iterable_value_t<PackedContainer>, Strategy>&& obs) const
167 {
168 if constexpr (std::same_as<TScheduler, schedulers::immediate>)
169 {
170 try
171 {
172 for (const auto& v : container)
173 {
174 if (obs.is_disposed())
175 return;
176
177 obs.on_next(v);
178 }
179
180 obs.on_completed();
181 }
182 catch (...)
183 {
184 obs.on_error(std::current_exception());
185 }
186 }
187 else
188 {
189 const auto worker = scheduler.create_worker();
190 obs.set_upstream(worker.get_disposable());
191 worker.schedule(from_iterable_schedulable{}, std::move(obs), container);
192 }
193 }
194};

303template<constraint::memory_model memory_model /* = memory_model::use_stack */, std::invocable<> Callable>
304auto from_callable(Callable&& callable)
305{
306 auto obs = just<memory_model>(std::forward<Callable>(callable));
307
308 if constexpr (std::same_as<utils::decayed_invoke_result_t<Callable>, void>)
309 return std::move(obs) | rpp::operators::map([](auto&& fn) { fn(); return utils::none{};});
310 else
311 return std::move(obs) | rpp::operators::map([](auto&& fn) { return fn(); });
312}
