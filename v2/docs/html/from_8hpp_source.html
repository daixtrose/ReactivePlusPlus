57 template<constraint::decayed_type PackedContainer>
58 struct from_iterable_strategy
59 {
60  RPP_NO_UNIQUE_ADDRESS PackedContainer container;
61 
62  template<constraint::observer_strategy<utils::iterable_value_t<PackedContainer>> Strategy>
63  void subscribe(const base_observer<utils::iterable_value_t<PackedContainer>, Strategy>& observer) const
64  {
65  try
66  {
67  for (const auto& v : container)
68  {
69  if (observer.is_disposed())
70  return;
71 
72  observer.on_next(v);
73  }
74 
75  observer.on_completed();
76  }
77  catch (...)
78  {
79  observer.on_error(std::current_exception());
80  }
81  }
82 };
