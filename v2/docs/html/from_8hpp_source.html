57 template<constraint::decayed_type Container>
58 class container_with_iterator
59 {
60 public:
61  template<typename ...Ts>
62  requires (!constraint::variadic_decayed_same_as<container_with_iterator<Container>, Ts...>)
63  explicit container_with_iterator(Ts&&...items)
64  : m_container{std::forward<Ts>(items)...} {}
65 
66  container_with_iterator(const container_with_iterator& other)
67  : m_container{other.m_container}
68  , m_index(other.m_index)
69  {}
70 
71  container_with_iterator(container_with_iterator&& other) noexcept
72  : m_container{std::move(other.m_container)}
73  , m_index(other.m_index)
74  {}
75 
76  auto begin() const { return std::cbegin(m_container); }
77  auto end() const { return std::cend(m_container); }
78 
79  auto get_actual_iterator() const
80  {
81  if (!m_iterator)
82  m_iterator = get_default_iterator_value();
83  return m_iterator.value();
84  }
85  bool increment_iterator() const
86  {
87  ++m_index;
88  return ++(m_iterator.value()) != end();
89  }
90 
91 private:
92  auto get_default_iterator_value() const
93  {
94  auto itr = begin();
95  std::advance(itr, m_index);
96  return itr;
97  }
98 
99 private:
100  RPP_NO_UNIQUE_ADDRESS Container m_container{};
101  mutable size_t m_index{};
102  mutable std::optional<decltype(std::cbegin(m_container))> m_iterator{};
103 };

120 template<constraint::decayed_type PackedContainer, schedulers::constraint::scheduler TScheduler>
121 struct from_iterable_strategy
122 {
123  RPP_NO_UNIQUE_ADDRESS PackedContainer container;
124  RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
125 
126  template<constraint::observer_strategy<utils::iterable_value_t<PackedContainer>> Strategy>
127  void subscribe(base_observer<utils::iterable_value_t<PackedContainer>, Strategy>&& observer) const
128  {
129  if constexpr (std::same_as<TScheduler, schedulers::immediate>)
130  {
131  try
132  {
133  for (const auto& v : container)
134  {
135  if (observer.is_disposed())
136  return;
137 
138  observer.on_next(v);
139  }
140 
141  observer.on_completed();
142  }
143  catch (...)
144  {
145  observer.on_error(std::current_exception());
146  }
147  }
148  else
149  {
150  const auto worker = scheduler.create_worker();
151  observer.set_upstream(worker.get_disposable());
152  worker.schedule([](const base_observer<utils::iterable_value_t<PackedContainer>, Strategy>& obs, const PackedContainer& cont) -> rpp::schedulers::optional_duration
153  {
154  try
155  {
156  if (const auto itr = cont.get_actual_iterator(); itr != std::cend(cont))
157  {
158  obs.on_next(utils::as_const(*itr));
159  if (cont.increment_iterator()) // it was not last
160  return schedulers::duration{}; // re-schedule this
161  }
162 
163  obs.on_completed();
164  }
165  catch(...)
166  {
167  obs.on_error(std::current_exception());
168  }
169  return std::nullopt;
170  }, std::move(observer), container);
171  }
172  }
173 };
