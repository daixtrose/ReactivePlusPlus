27template<typename TScheduler>
28struct interval_strategy
29{
30 using ValueType = size_t;
31
32 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
33 rpp::schedulers::duration initial;
34 rpp::schedulers::duration period;
35
36 template<rpp::constraint::observer_of_type<ValueType> TObs>
37 void subscribe(TObs&& observer) const
38 {
39 const auto worker = scheduler.create_worker();
40 observer.set_upstream(worker.get_disposable());
41 worker.schedule(initial, interval_schedulable{}, std::forward<TObs>(observer), period, size_t{});
42 }
43};
44}
