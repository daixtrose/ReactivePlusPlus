102template<constraint::decayed_type Type, constraint::observer_strategy<Type> Strategy, typename DisposablesStrategy>
103class observer_impl
104{
105protected:
106 template<typename... Args>
107 requires constraint::is_constructible_from<Strategy, Args&&...>
108 explicit observer_impl(DisposablesStrategy&& strategy, Args&&... args)
109 : m_strategy{std::forward<Args>(args)...}, m_disposable{std::move(strategy)}
110 {
111 }
112
113 observer_impl(const observer_impl&) = default;
114 observer_impl(observer_impl&&) noexcept = default;
115
116public:
117 using on_next_lvalue = void(observer_impl::*)(const Type&) const noexcept;
118 using on_next_rvalue = void(observer_impl::*)(Type&&) const noexcept;
123 void set_upstream(const disposable_wrapper& d)
124 {
125 if (is_disposed()) {
126 d.dispose();
127 return;
128 }
129
130 m_disposable.set_upstream(d);
131 m_strategy.set_upstream(d);
132 }
133
140 bool is_disposed() const noexcept
141 {
142 return m_disposable.is_disposed() || m_strategy.is_disposed();
143 }
149 void on_next(const Type& v) const noexcept
150 {
151 try
152 {
153 if (!is_disposed())
154 m_strategy.on_next(v);
155 }
156 catch(...)
157 {
158 on_error(std::current_exception());
159 }
160 }
161
167 void on_next(Type&& v) const noexcept
168 {
169 try
170 {
171 if (!is_disposed())
172 m_strategy.on_next(std::move(v));
173 }
174 catch(...)
175 {
176 on_error(std::current_exception());
177 }
178 }
179
185 void on_error(const std::exception_ptr& err) const noexcept
186 {
187 if (!is_disposed())
188 {
189 rpp::utils::finally_action finally{[&]{ m_disposable.dispose(); }};
190 m_strategy.on_error(err);
191 }
192 }
193
198 void on_completed() const noexcept
199 {
200 if (!is_disposed())
201 {
202 rpp::utils::finally_action finally{[&]{ m_disposable.dispose(); }};
203 m_strategy.on_completed();
204 }
205 }
206
207private:
208 RPP_NO_UNIQUE_ADDRESS Strategy m_strategy;
209 RPP_NO_UNIQUE_ADDRESS DisposablesStrategy m_disposable;
210};
