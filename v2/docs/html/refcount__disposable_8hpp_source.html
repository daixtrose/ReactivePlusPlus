30{
31public:
32 refcount_disposable() = default;
33 refcount_disposable(disposable_ptr target) { m_underlying.add(std::move(target)); }
34

37
38 bool is_disposed() const noexcept override { return m_refcount.load(std::memory_order_acquire) == 0; }
39
40 void dispose() override
41 {
42 while (auto current_value = m_refcount.load(std::memory_order_acquire))
43 {
44 if (!m_refcount.compare_exchange_strong(current_value, current_value - 1, std::memory_order_acq_rel))
45 continue;
46
47 // was last one
48 if (current_value == 1)
49 m_underlying.dispose();
50
51 return;
52 }
53 }
54
55 composite_disposable_wrapper add_ref()
56 {
57 while (auto current_value = m_refcount.load(std::memory_order_acquire))
58 {
59 if (!m_refcount.compare_exchange_strong(current_value, current_value + 1, std::memory_order_acq_rel))
60 continue;
61
62 return composite_disposable_wrapper{shared_from_this()};
63 }
64 return composite_disposable_wrapper{};
65 }
66
67 using interface_composite_disposable::add;
68
69 void add(disposable_wrapper disposable) override
70 {
71 m_underlying.add(std::move(disposable));
72 }
73
74private:
75 composite_disposable m_underlying;
76 std::atomic_size_t m_refcount{1};
77};
