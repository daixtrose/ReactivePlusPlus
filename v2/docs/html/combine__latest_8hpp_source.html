26template<rpp::constraint::observer Observer, typename TSelector, rpp::constraint::decayed_type... Args>
27class combine_latest_disposable final : public composite_disposable
28{
29public:
30 explicit combine_latest_disposable(Observer&& observer, const TSelector& selector)
31 : observer_with_mutex{std::move(observer)}
32 , selector{selector}
33 {
34 }
35
36 pointer_under_lock<Observer> get_observer_under_lock() { return pointer_under_lock{observer_with_mutex}; }
37
38 rpp::utils::tuple<std::optional<Args>...>& get_values() { return values; }
39
40 const TSelector& get_selector() const { return selector; }
41
42 bool decrement_on_completed()
43 {
44 // just need atomicity, not guarding anything
45 return m_on_completed_needed.fetch_sub(1, std::memory_order::seq_cst) == 1;
46 }
47
48private:
49 value_with_mutex<Observer> observer_with_mutex{};
50 rpp::utils::tuple<std::optional<Args>...> values{};
51 RPP_NO_UNIQUE_ADDRESS TSelector selector;
52
53 std::atomic_size_t m_on_completed_needed{sizeof...(Args)};
54};

56template<size_t I, rpp::constraint::observer Observer, typename TSelector, rpp::constraint::decayed_type... Args>
57struct combine_latest_observer_strategy
58{
59 std::shared_ptr<combine_latest_disposable<Observer, TSelector, Args...>> disposable{};
60
61 void set_upstream(const rpp::disposable_wrapper& d) const
62 {
63 disposable->add(d);
64 }
65
66 bool is_disposed() const
67 {
68 return disposable->is_disposed();
69 }
70
71 template<typename T>
72 void on_next(T&& v) const
73 {
74 // mutex need to be locked during changing of values, generating new values and sending of new values due to we can't update value while we are sending old one
75 const auto observer = disposable->get_observer_under_lock();
76 disposable->get_values().template get<I>().emplace(std::forward<T>(v));
77
78 disposable->get_values().apply([this, &observer](const std::optional<Args>&... vals) {
79 if ((vals.has_value() && ...))
80 observer->on_next(disposable->get_selector()(vals.value()...));
81 });
82 }
83
84 void on_error(const std::exception_ptr& err) const
85 {
86 disposable->dispose();
87 disposable->get_observer_under_lock()->on_error(err);
88 }
89
90 void on_completed() const
91 {
92 if (disposable->decrement_on_completed())
93 {
94 disposable->dispose();
95 disposable->get_observer_under_lock()->on_completed();
96 }
97 }
98};

100template<typename TSelector, rpp::constraint::observable... TObservables>
101struct combine_latest_t
102{
103 RPP_NO_UNIQUE_ADDRESS rpp::utils::tuple<TObservables...> observables;
104 RPP_NO_UNIQUE_ADDRESS TSelector selector;
105
106 template<rpp::constraint::decayed_type T>
107 requires std::invocable<TSelector, T, rpp::utils::extract_observable_type_t<TObservables>...>
108 using result_value = std::invoke_result_t<TSelector, T, rpp::utils::extract_observable_type_t<TObservables>...>;
109
110 template<rpp::details::observables::constraint::disposable_strategy Prev>
111 using updated_disposable_strategy = rpp::details::observables::fixed_disposable_strategy_selector<1>;
112
113 template<rpp::constraint::observer Observer, typename... Strategies>
114 void subscribe(Observer&& observer, const observable_chain_strategy<Strategies...>& observable_strategy) const
115 {
116 // Need to take ownership over current_thread in case of inner-observables also uses them
117 auto drain_on_exit = rpp::schedulers::current_thread::own_queue_and_drain_finally_if_not_owned();
118 observables.apply(&subscribe_impl<Observer, Strategies...>, std::forward<Observer>(observer), observable_strategy, selector);
119 }
120
121private:
122 template<rpp::constraint::observer Observer, typename... Strategies>
123 static void subscribe_impl(Observer&& observer, const observable_chain_strategy<Strategies...>& observable_strategy, const TSelector& selector, const TObservables&... observables)
124 {
125 using ExpectedValue = typename observable_chain_strategy<Strategies...>::value_type;
126 using Disposable = combine_latest_disposable<Observer, TSelector, ExpectedValue, rpp::utils::extract_observable_type_t<TObservables>...>;
127
128 auto disposable = std::make_shared<Disposable>(std::forward<Observer>(observer), selector);
129 disposable->get_observer_under_lock()->set_upstream(rpp::disposable_wrapper::from_weak(disposable));
130 subscribe<std::decay_t<ExpectedValue>>(disposable, std::index_sequence_for<TObservables...>{}, observables...);
131
132 observable_strategy.subscribe(rpp::observer<ExpectedValue, combine_latest_observer_strategy<0, std::decay_t<Observer>, TSelector, ExpectedValue, rpp::utils::extract_observable_type_t<TObservables>...>>{std::move(disposable)});
133 }
134
135 template<typename ExpectedValue, rpp::constraint::observer Observer, size_t... I>
136 static void subscribe(std::shared_ptr<combine_latest_disposable<Observer, TSelector, ExpectedValue, rpp::utils::extract_observable_type_t<TObservables>...>> disposable, std::index_sequence<I...>, const TObservables&... observables)
137 {
138 (..., observables.subscribe(rpp::observer<rpp::utils::extract_observable_type_t<TObservables>, combine_latest_observer_strategy<I + 1, Observer, TSelector, ExpectedValue, rpp::utils::extract_observable_type_t<TObservables>...>>{disposable}));
139 }
140};
