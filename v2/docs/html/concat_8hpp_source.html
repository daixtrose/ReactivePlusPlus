33 RPP_NO_UNIQUE_ADDRESS mutable TObserver observer;
34 RPP_NO_UNIQUE_ADDRESS mutable PackedContainer container;

64template<rpp::schedulers::constraint::worker TWorker, rpp::constraint::observer TObserver, typename PackedContainer>
65void drain(TObserver&& obs, const TWorker& worker, PackedContainer&& container, size_t index)
66{
67 std::optional<utils::iterable_value_t<PackedContainer>> observable{};
68 bool is_last_observable = false;
69 try
70 {
71 auto itr = std::cbegin(container);
72 std::advance(itr, static_cast<int64_t>(index++));
73
74 if (itr != std::cend(container))
75 {
76 observable.emplace(*itr);
77 is_last_observable = std::next(itr) == std::cend(container);
78 }
79 }
80 catch (...)
81 {
82 obs.on_error(std::current_exception());
83 return;
84 }
85
86 if (!observable.has_value())
87 {
88 obs.on_completed();
89 return;
90 }
91
92 using ValueType = rpp::utils::extract_observable_type_t<utils::iterable_value_t<PackedContainer>>;
93
94 if (is_last_observable)
95 observable->subscribe(std::forward<TObserver>(obs));
96 else
97 observable->subscribe(observer<ValueType, concat_source_observer_strategy<TWorker, std::decay_t<TObserver>, std::decay_t<PackedContainer>>>{std::forward<TObserver>(obs), std::forward<PackedContainer>(container), worker, index});
98 return;
99}

101template<rpp::schedulers::constraint::scheduler TScheduler, constraint::decayed_type PackedContainer>
102struct concat_strategy
103{
104 template<typename... Args>
105 requires (!constraint::variadic_decayed_same_as<concat_strategy<TScheduler, PackedContainer>, Args...>)
106 concat_strategy(const TScheduler& scheduler, Args&&... args)
107 : container{std::forward<Args>(args)...}
108 , scheduler{scheduler}
109 {
110 }
111
112 RPP_NO_UNIQUE_ADDRESS PackedContainer container;
113 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
114
115 using ValueType = rpp::utils::extract_observable_type_t<utils::iterable_value_t<PackedContainer>>;
116
117 template<constraint::observer_strategy<ValueType> Strategy>
118 void subscribe(observer<ValueType, Strategy>&& obs) const
119 {
120 const auto worker = scheduler.create_worker();
121 if (auto d = worker.get_disposable(); !d.is_disposed())
122 obs.set_upstream(std::move(d));
123 drain(std::move(obs), worker, container, size_t{});
124 }
125};

200template<constraint::memory_model MemoryModel /*= memory_model::use_stack*/, rpp::constraint::observable TObservable, rpp::constraint::observable... TObservables, rpp::schedulers::constraint::scheduler TScheduler>
201 requires (std::same_as<rpp::utils::extract_observable_type_t<TObservable>, rpp::utils::extract_observable_type_t<TObservables>> && ...)
202auto concat(const TScheduler& scheduler, TObservable&& obs, TObservables&&... others)
203{
204 if constexpr ((rpp::constraint::decayed_same_as<TObservable, TObservables> && ...))
205 {
206 using inner_container = std::array<std::decay_t<TObservable>, sizeof...(TObservables) + 1>;
207 using container = std::conditional_t<std::same_as<MemoryModel, rpp::memory_model::use_stack>, inner_container, details::shared_container<inner_container>>;
208 return rpp::details::make_concat_from_iterable<container>(scheduler, std::forward<TObservable>(obs), std::forward<TObservables>(others)...);
209 }
210 else
211 return concat<MemoryModel>(scheduler, std::forward<TObservable>(obs).as_dynamic(), std::forward<TObservables>(others).as_dynamic()...);
212}
