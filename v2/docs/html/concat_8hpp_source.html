60template<constraint::decayed_type PackedContainer>
61struct concat_strategy
62{
63 RPP_NO_UNIQUE_ADDRESS PackedContainer container;
64
65 using Type = utils::extract_observable_type_t<utils::iterable_value_t<PackedContainer>>;
66
67 template<constraint::observer_strategy<Type> Strategy>
68 void subscribe(observer<Type, Strategy>&& obs) const
69 {
70 drain(container, std::move(obs));
71 }
72
73 template<constraint::observer_strategy<Type> Strategy>
74 static void drain(constraint::decayed_same_as<PackedContainer> auto&& container, observer<Type, Strategy>&& obs)
75 {
76 if (const auto itr = container.get_actual_iterator(); itr != std::cend(container))
77 {
78 decltype(auto) observable = PackedContainer::extract_value_from_itr(itr);
79 observable.subscribe(observer<Type,
80 rpp::operators::details::operator_strategy_base<Type, observer<Type, Strategy>,
81 concat_source_observer_strategy<PackedContainer>>>
82 {
83 std::move(obs),
84 std::forward<decltype(container)>(container)
85 });
86 }
87 else
88 {
89 obs.on_completed();
90 }
91 }
92};
