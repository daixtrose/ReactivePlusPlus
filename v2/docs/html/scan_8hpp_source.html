23template<rpp::constraint::decayed_type Seed, rpp::constraint::decayed_type Fn>
24struct scan_observer_strategy
25{
26 RPP_NO_UNIQUE_ADDRESS mutable Seed seed;
27 RPP_NO_UNIQUE_ADDRESS Fn fn;
28
29 void on_subscribe(const rpp::constraint::observer auto& obs) const
30 {
31 obs.on_next(seed);
32 }
33
34 template<typename T>
35 void on_next(const rpp::constraint::observer auto& obs, T&& v) const
36 {
37 seed = fn(std::move(seed), std::forward<T>(v));
38 obs.on_next(utils::as_const(seed));
39 }
40
41 constexpr static forwarding_on_error_strategy on_error{};
42 constexpr static forwarding_on_completed_strategy on_completed{};
43 constexpr static forwarding_set_upstream_strategy set_upstream{};
44 constexpr static forwarding_is_disposed_strategy is_disposed{};
45
46};

73template<rpp::constraint::decayed_type Seed, rpp::constraint::decayed_type Fn>
74struct scan_no_seed_observer_strategy
75{
76 RPP_NO_UNIQUE_ADDRESS Fn fn;
77 mutable std::optional<Seed> seed{};
78
79 template<rpp::constraint::decayed_same_as<Seed> T>
80 void on_next(const rpp::constraint::observer auto& obs, T&& v) const
81 {
82 if (seed)
83 seed = fn(std::move(seed).value(), std::forward<T>(v));
84 else
85 seed = std::forward<T>(v);
86
87 obs.on_next(utils::as_const(seed.value()));
88 }
89
90 constexpr static forwarding_on_error_strategy on_error{};
91 constexpr static forwarding_on_completed_strategy on_completed{};
92 constexpr static forwarding_set_upstream_strategy set_upstream{};
93 constexpr static forwarding_is_disposed_strategy is_disposed{};
94 constexpr static empty_on_subscribe on_subscribe{};
95};
