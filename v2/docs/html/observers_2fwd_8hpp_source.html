30template<typename S, typename Type>
31concept observer_strategy = requires(const S& const_strategy, S& strategy, const Type& v, const rpp::composite_disposable& disposable)
32{
33 const_strategy.on_next(v);
34 const_strategy.on_next(Type{});
35 const_strategy.on_error(std::exception_ptr{});
36 const_strategy.on_completed();
37
38 strategy.set_upstream(disposable);
39 { strategy.is_disposed() } -> std::same_as<bool>;
40};
