32template<typename S, typename Type>
33concept observer_strategy = requires(const S& const_strategy, S& strategy, const Type& v, Type& mv, const rpp::disposable_wrapper& disposable)
34{
35 const_strategy.on_next(v);
36 const_strategy.on_next(std::move(mv));
37 const_strategy.on_error(std::exception_ptr{});
38 const_strategy.on_completed();
39
40 strategy.set_upstream(disposable);
41 { strategy.is_disposed() } -> std::same_as<bool>;
42};
43} // namespace rpp::constraint
