35  static void sleep_until(const time_point timepoint)
36  {
37  if (timepoint <= s_last_now_time)
38  return;
39 
40  const auto now = clock_type::now();
41  std::this_thread::sleep_for(timepoint - now);
42  s_last_now_time = std::max(now, timepoint);
43  }
44 
45  static time_point get_now() { return s_last_now_time = clock_type::now(); }

47  static void drain_queue(std::optional<details::schedulables_queue>& queue)
48  {
49  while (!queue->is_empty())
50  {
51  auto top = queue->pop();
52  if (top->is_disposed())
53  continue;
54 
55  sleep_until(top->get_timepoint());
56 
57  optional_duration duration{0};
58  // immediate like scheduling
59  do
60  {
61  if (duration.value() > duration::zero() && !top->is_disposed())
62  std::this_thread::sleep_for(duration.value());
63 
64  if (top->is_disposed())
65  duration.reset();
66  else
67  duration = (*top)();
68 
69  } while (queue->is_empty() && duration.has_value());
70 
71  if (duration.has_value())
72  queue->emplace(get_now() + duration.value(), std::move(top));
73  }
74 
75  queue.reset();
76  }

83  static void defer_for(duration duration, Fn&& fn, TObs&& obs, Args&&... args)
84  {
85  auto& queue = s_queue;
86  const bool someone_owns_queue = queue.has_value();
87  if (!someone_owns_queue)
88  {
89  queue.emplace();
90 
91  const auto optional_duration = details::immediate_scheduling_while_condition(duration, [&queue](){ return queue->is_empty(); }, fn, obs, args...);
92  if (!optional_duration || obs.is_disposed())
93  return drain_queue(queue);
94  duration = optional_duration.value();
95  }
96  else if (obs.is_disposed())
97  return;
98 
99  queue->emplace(get_now() + duration, std::forward<Fn>(fn), std::forward<TObs>(obs), std::forward<Args>(args)...);
100 
101  if (!someone_owns_queue)
102  drain_queue(queue);
103  }
