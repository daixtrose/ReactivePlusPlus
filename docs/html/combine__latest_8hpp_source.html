49 template<size_t I>
50 struct combine_latest_on_next
51 {
52  template<typename TCombiner, constraint::decayed_type... Types>
53  void operator()(auto&& value,
54  const auto& subscriber,
55  const std::shared_ptr<combine_latest_state<TCombiner, Types...>>& state) const
56  {
57  // mutex need to be locked during changing of values, generating new values and sending of new values due to we can't update value while we are sending old one
58  std::scoped_lock lock{state->values_mutex};
59  std::get<I>(state->values) = std::forward<decltype(value)>(value);
60 
61  std::apply([&](const auto&...cached_values)
62  {
63  if ((cached_values.has_value() && ...))
64  subscriber.on_next(state->combiner(cached_values.value()...));
65  },
66  state->values);
67  }
68 };

100  template<size_t...I>
101  void subscribe_other_observables(std::index_sequence<I...>,
102  // Used in compile time for variadic expansion
103  const auto& subscriber,

119  static auto create_inner_subscriber(auto&& subscriber,
120  std::shared_ptr<combine_latest_state<TCombiner, Type, utils::extract_observable_type_t<TOtherObservable>...>> state)
121  {
122  auto subscription = state->children_subscriptions.make_child();
123  return create_subscriber_with_state<ValueType>(std::move(subscription),
124  combine_latest_on_next<I>{},

126  combine_latest_on_completed{},

128  std::move(state));

138  auto state = std::make_shared<combine_latest_state_with_serialized_spinlock<TCombiner, Type, utils::extract_observable_type_t<TOtherObservable>...>>(m_combiner, in_subscriber.get_subscription());
139  // change subscriber to serialized to avoid manual using of mutex
140  auto subscriber = make_serialized_subscriber(std::forward<TSub>(in_subscriber), std::shared_ptr<utils::spinlock>{state, &state->spinlock});
141 
142  state->count_of_on_completed_needed.store(sizeof...(TOtherObservable) + 1, std::memory_order::relaxed);
143 
144  // Subscribe to other observables and redirect on_next event to state
145  subscribe_other_observables(std::index_sequence_for<TOtherObservable...>{}, subscriber, state);
146 
147  // Redirect values from this observable to the state for value composition
148  return create_inner_subscriber<Type, s_index_of_source_type>(std::move(subscriber), std::move(state));
