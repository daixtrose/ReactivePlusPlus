43 void iterate(const auto& iterable,
44  const schedulers::constraint::scheduler auto& scheduler,
45  const constraint::subscriber auto& subscriber)
46 {
47  if constexpr (constraint::decayed_same_as<decltype(scheduler), schedulers::immediate>)
48  {
49  for (const auto& v : extract_iterable_from_packed(iterable))
50  {
51  if (subscriber.is_subscribed())
52  subscriber.on_next(utils::as_const(v));
53  else
54  return;
55  }
56  subscriber.on_completed();
57  }
58  else
59  {
60  auto worker = scheduler.create_worker(subscriber.get_subscription());
61  worker.schedule([=, index = size_t{0}]() mutable-> schedulers::optional_duration
62  {
63  if (subscriber.is_subscribed())
64  {
65  const auto& extracted_iterable = extract_iterable_from_packed(iterable);
66  const auto end = std::cend(extracted_iterable);
67  auto itr = std::cbegin(extracted_iterable);
68 
69  std::ranges::advance(itr, static_cast<int64_t>(index), end);
70 
71  if (itr != end)
72  {
73  subscriber.on_next(utils::as_const(*itr));
74  if (std::next(itr) != end) // it was not last
75  {
76  ++index;
77  return schedulers::duration{}; // re-schedule this
78  }
79  }
80 
81  subscriber.on_completed();
82  }
83  return schedulers::optional_duration{};
84  });
85  }
