39 template<constraint::decayed_type T>
40 struct dynamic_observer_callables
41 {
42 template<constraint::on_next_fn<T> OnNext = utils::empty_function_t<T>,
43 constraint::on_error_fn OnError = utils::empty_function_t<std::exception_ptr>,
44 constraint::on_completed_fn OnCompleted = utils::empty_function_t<>>
45 dynamic_observer_callables(OnNext&& on_next = {}, OnError&& on_error = {}, OnCompleted&& on_completed = {})
46 : m_state{make_shared_state(std::forward<OnNext>(on_next),
47 std::forward<OnError>(on_error),
48 std::forward<OnCompleted>(on_completed))} {}
49
50 dynamic_observer_callables(const std::shared_ptr<interface_observer<T>>& state)
51 : m_state{state}{}
52
53 const interface_observer<T>* operator->() const { return m_state.get(); }
54
55 private:
56 template<typename ...Args>
57 static auto make_shared_state(Args&&...args)
58 {
59 return std::make_shared<specific_observer<T, std::decay_t<Args>...>>(std::forward<Args>(args)...);
60 }
61
62 std::shared_ptr<interface_observer<T>> m_state;
63 };
