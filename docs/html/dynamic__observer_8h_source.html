42template<typename T>
43class dynamic_observer final : public interface_observer<T>
44{
45 static_assert(std::is_same_v<std::decay_t<T>, T>, "Type should be decayed to match with decayed observable types");
46
47public:
48 template<typename OnNext = utils::empty_function_t<T>,
49 typename OnError = utils::empty_function_t<std::exception_ptr>,
50 typename OnCompleted = utils::empty_function_t<>,
51 typename = utils::enable_if_observer_constructible_t<T, OnNext, OnError, OnCompleted>>
52 dynamic_observer(OnNext&& on_next = {}, OnError&& on_error = {}, OnCompleted&& on_completed = {})
53 : m_state{make_shared_state(std::forward<OnNext>(on_next),
54 std::forward<OnError>(on_error),
55 std::forward<OnCompleted>(on_completed))} { }
56
57 template<typename OnNext,
58 typename OnCompleted,
59 typename = utils::enable_if_observer_constructible_t<T, OnNext, utils::empty_function_t<std::exception_ptr>, OnCompleted>>
60 dynamic_observer(OnNext&& on_next, OnCompleted&& on_completed)
61 : dynamic_observer{std::forward<OnNext>(on_next),
62 utils::empty_function_t<std::exception_ptr>{},
63 std::forward<OnCompleted>(on_completed)} {}
64
65 template<typename TObserver,
66 typename = std::enable_if_t<utils::is_observer_v<TObserver> && !std::is_same_v<std::decay_t<TObserver>, dynamic_observer<T>>>>
67 dynamic_observer(TObserver&& obs)
68 : m_state{std::make_shared<std::decay_t<TObserver>>(std::forward<TObserver>(obs))} {}
69
70
71 dynamic_observer(const dynamic_observer<T>&) = default;
72 dynamic_observer(dynamic_observer<T>&&) noexcept = default;
73
74 void on_next(const T& v) const override { m_state->on_next(v); }
75 void on_next(T&& v) const override { m_state->on_next(std::move(v)); }
76 void on_error(const std::exception_ptr& err) const override { m_state->on_error(err); }
77 void on_completed() const override { m_state->on_completed(); }
78
79 dynamic_observer<T> as_dynamic() const { return *this; }
80
81private:
82 template<typename ...Args>
83 static auto make_shared_state(Args&&...args)
84 {
85 return std::make_shared<specific_observer<T, std::decay_t<Args>...>>(std::forward<Args>(args)...);
86 }
87
88 std::shared_ptr<interface_observer<T>> m_state;
89};
