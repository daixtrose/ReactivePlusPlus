44template<constraint::decayed_type Type, constraint::on_subscribe_fn<Type> OnSubscribeFn>
45class specific_observable final : public interface_observable<Type, specific_observable<Type, OnSubscribeFn>>
46{
47public:
48 specific_observable(constraint::decayed_same_as<OnSubscribeFn> auto&& on_subscribe)
49 : m_state{std::forward<decltype(on_subscribe)>(on_subscribe)} {}
50
54 [[nodiscard]] auto as_dynamic() const &
55 {
56 return rpp::dynamic_observable<Type>{*this};
57 }
58
62 [[nodiscard]] auto as_dynamic() &&
63 {
64 return rpp::dynamic_observable<Type>{std::move(*this)};
65 }
66
67 subscription subscribe(const dynamic_subscriber<Type>& subscriber) const noexcept override
68 {
69 return subscribe_impl(subscriber);
70 }

77 template<constraint::subscriber_of_type<Type> TSub>
78 subscription subscribe(TSub&& subscriber) const noexcept
79 {
80 return subscribe_impl(std::forward<TSub>(subscriber));
81 }
82
88 template<constraint::observer_of_type<Type> TObserver>
89 subscription subscribe(TObserver&& observer) const noexcept
90 {
91 return subscribe_impl<std::decay_t<TObserver>>(std::forward<TObserver>(observer));
92 }
93
99 template<typename ...Args>
100 requires std::is_constructible_v<dynamic_subscriber<Type>, std::decay_t<Args>...>
101 subscription subscribe(Args&&...args) const noexcept
102 {
103 return subscribe_impl(rpp::make_specific_subscriber<Type>(std::forward<Args>(args)...));
104 }
105
106private:
107 template<typename Obs>
108 subscription subscribe_impl(const specific_subscriber<Type, Obs>& subscriber) const noexcept
109 {
110 try
111 {
112 m_state(subscriber);
113 }
114 catch (const std::exception& exc)
115 {
116 subscriber.on_error(std::make_exception_ptr(exc));
117 }
118 return subscriber.get_subscription();
119 }
120private:
121 OnSubscribeFn m_state;
122};
