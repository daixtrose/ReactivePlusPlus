100private:
101 template<constraint::observer_of_type<Type> Obs>
102 composite_subscription subscribe_impl(const specific_subscriber<Type, Obs>& subscriber) const noexcept
103 {
104 try
105 {
106 m_state(subscriber);
107 }
108 catch (const std::exception& exc)
109 {
110 subscriber.on_error(std::make_exception_ptr(exc));
111 }
112 return subscriber.get_subscription();
113 }
114private:
115 OnSubscribeFn m_state;
116};
