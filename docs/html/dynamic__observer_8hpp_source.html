143template<constraint::decayed_type T>
144class dynamic_observer final : public details::dynamic_state_observer<T>
145{
146public:
147 template<constraint::on_next_fn<T> OnNext = utils::empty_function_t<T>,
148 constraint::on_error_fn OnError = utils::rethrow_error_t,
149 constraint::on_completed_fn OnCompleted = utils::empty_function_t<>>
150 dynamic_observer(OnNext&& on_next = {}, OnError&& on_error = {}, OnCompleted&& on_completed = {})
151 : details::dynamic_state_observer<T>{std::forward<OnNext>(on_next),
152 std::forward<OnError>(on_error),
153 std::forward<OnCompleted>(on_completed)} {}
154
155 dynamic_observer(constraint::on_next_fn<T> auto&& on_next, constraint::on_completed_fn auto&& on_completed)
156 : details::dynamic_state_observer<T>{std::forward<decltype(on_next)>(on_next),
157 utils::rethrow_error_t{},
158 std::forward<decltype(on_completed)>(on_completed)} {}
159
160 template<constraint::observer_of_type<T> TObserver>
161 requires (!std::is_same_v<std::decay_t<TObserver>, dynamic_observer<T>>)
162 dynamic_observer(TObserver&& obs)
163 : details::dynamic_state_observer<T>{std::forward<TObserver>(obs)} {}
164
168 const dynamic_observer<T>& as_dynamic() const { return *this; }
169};
