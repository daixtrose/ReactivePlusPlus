34template<constraint::decayed_type Type, typename OnNext, typename OnError, typename OnCompleted>
35static auto make_lift_action_by_callbacks(OnNext&& on_next, OnError&& on_error, OnCompleted&& on_completed)
36{
37 return [on_next = std::forward<OnNext>(on_next),
38 on_error = std::forward<OnError>(on_error),
39 on_completed = std::forward<OnCompleted>(on_completed)]<constraint::subscriber TSub>(TSub&& subscriber)
40 {
41 auto subscription = subscriber.get_subscription();
42 return specific_subscriber<Type, state_observer<Type,
43 std::decay_t<TSub>,
44 std::decay_t<OnNext>,
45 std::decay_t<OnError>,
46 std::decay_t<OnCompleted>>>
47 {
48 subscription,
49 std::forward<TSub>(subscriber),
50 on_next,
51 on_error,
52 on_completed
53 };
54 };
