48template<constraint::decayed_type Type, typename OnNext, typename OnError, typename OnCompleted>
49static auto make_lift_action_by_callbacks(OnNext&& on_next, OnError&& on_error, OnCompleted&& on_completed)
50{
51 return [on_next = std::forward<OnNext>(on_next),
52 on_error = std::forward<OnError>(on_error),
53 on_completed = std::forward<OnCompleted>(on_completed)]<constraint::subscriber TSub>(TSub&& subscriber)
54 {
55 auto subscription = subscriber.get_subscription();
56 return specific_subscriber<Type, state_observer<Type,
57 std::decay_t<TSub>,
58 std::decay_t<OnNext>,
59 std::decay_t<OnError>,
60 std::decay_t<OnCompleted>>>
61 {
62 subscription,
63 std::forward<TSub>(subscriber),
64 on_next,
65 on_error,
66 on_completed
67 };
68 };

248 template<constraint::decayed_type NewType, details::lift_fn<NewType> OperatorFn, typename FwdThis>
249 static auto lift_impl(OperatorFn&& op, FwdThis&& _this)
250 {
251 auto action = [new_this = std::forward<FwdThis>(_this), op = std::forward<OperatorFn>(op)](auto&& subscriber)
252 {
253 new_this.subscribe(op(std::forward<decltype(subscriber)>(subscriber)));
254 };
255
256 return specific_observable<NewType, decltype(action)>(std::move(action));
