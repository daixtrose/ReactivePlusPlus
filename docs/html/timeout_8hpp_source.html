38 template<typename Worker>
39 struct timeout_on_next
40 {
41  template<typename Value>
42  void operator()(Value&& v, const auto& subscriber, const std::shared_ptr<timeout_state>& state) const
43  {
44  if (state->last_emission_time.exchange(Worker::now(), std::memory_order_acq_rel) != timeout_state::s_timeout_reached)
45  subscriber.on_next(std::forward<Value>(v));
46  }

60 template<constraint::decayed_type Type, schedulers::constraint::scheduler TScheduler>
61 struct timeout_impl
62 {
63  schedulers::duration period;
64  TScheduler scheduler;
65 
66  template<constraint::subscriber_of_type<Type> TSub>
67  auto operator()(TSub&& in_subscriber) const
68  {
69  auto state = std::make_shared<timeout_state_with_serialized_spinlock>(in_subscriber.get_subscription());
70  // change subscriber to serialized to avoid manual using of mutex
71  auto subscriber = make_serialized_subscriber(std::forward<TSub>(in_subscriber),
72  std::shared_ptr<utils::spinlock>{state, &state->spinlock});
73 
74  const auto worker = scheduler.create_worker(state->children_subscriptions);
75  state->last_emission_time.store(worker.now(), std::memory_order_relaxed);
76 
77  const auto last_emission_time = state->last_emission_time.load(std::memory_order_relaxed);
78  worker.schedule(last_emission_time + period,
79  [period = period, prev_emission_time = last_emission_time, subscriber, state]() mutable -> schedulers::optional_duration
80  {
81  while (true)
82  {
83  // last emission time still same value -> timeout reached, else -> prev_emission_time
84  // would be update to actual emission time
85  if (state->last_emission_time.compare_exchange_strong(prev_emission_time,
86  timeout_state::s_timeout_reached,
87  std::memory_order_acq_rel))
88  return time_is_out(state, subscriber);
89 
90  // if we still need to wait a bit more -> let's wait
91  if (const auto diff_to_schedule = (prev_emission_time + period) - decltype(worker)::now();
92  diff_to_schedule > rpp::schedulers::duration{0})
93  return diff_to_schedule;
94 
95  // okay, we here because:
96  // 1) last_emission_time was not equal to prev_emission_time
97  // 2) last_emission_time + period before now -> we are still in timeout state
98  // 3) prev_emission_time updated to last_emission_time
99  // So we can return to begin
100  }
101  });
102 
103  return create_subscriber_with_state<Type>(state->children_subscriptions,
104  timeout_on_next<decltype(worker)>{},
105  timeout_on_error{},
106  timeout_on_completed{},
107  std::move(subscriber),
108  std::move(state));
109  }
110 
111 private:
112  static schedulers::optional_duration time_is_out(const auto& state, const auto& subscriber)
113  {
114  state->children_subscriptions.unsubscribe();
115  subscriber.on_error(std::make_exception_ptr(utils::timeout{"Timeout reached"}));
116  return std::nullopt;
117  }
118 };
